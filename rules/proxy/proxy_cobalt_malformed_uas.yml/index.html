title: CobaltStrike Malformed UAs in Malleable Profiles

detection:
  selection1:
    c-useragent: 
      - 'Mozilla/4.0 (compatible; MSIE 6.0;Windows NT 5.1)'

  selection2:
    c-useragent|endswith: '; MANM; MANM)'
  condition: 1 of selection*
