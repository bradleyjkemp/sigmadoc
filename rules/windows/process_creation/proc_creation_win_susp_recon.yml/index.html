title: Recon Information for Export with Command Prompt

logsource:
    product: windows
    category: process_creation
detection:
    selection_image:
        Image|endswith:
            - '\tree.com'
            - '\WMIC.exe'
            - '\doskey.exe'
            - '\sc.exe'
    selection_redirect:
        ParentCommandLine|contains: ' > %TEMP%\'
    condition: selection_image and selection_redirect

    - Unknown
