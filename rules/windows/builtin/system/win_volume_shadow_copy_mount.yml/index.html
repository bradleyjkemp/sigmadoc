title: Volume Shadow Copy Mount

    - attack.credential_access

logsource:
    product: windows
    service: system
detection:
    selection: 
        Provider_Name: Microsoft-Windows-Ntfs
        EventID: 98
        DeviceName|contains: HarddiskVolumeShadowCopy
    condition: selection
falsepositives:
    - Legitimate use of volume shadow copy mounts (backups maybe).
